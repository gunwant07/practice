const express = require('express');
const cors = require('cors');
const bodyParser = require('body-parser');
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken'); // (still unused, but kept)
require('dotenv').config();

const userRoutes = require('./routes/userRoutes');
const authRoutes = require('./routes/authRoutes');
const db = require('./models/db'); // MySQL pool connection

// ðŸ”¹ Prometheus client setup
const client = require('prom-client');
const register = new client.Registry();
client.collectDefaultMetrics({ register });

const httpRequestCounter = new client.Counter({
  name: 'http_requests_total',
  help: 'Total number of HTTP requests',
  labelNames: ['method', 'route', 'status'],
});
register.registerMetric(httpRequestCounter);

const app = express();
const PORT = process.env.PORT || 5000;

// Middleware
app.use(cors());
app.use(bodyParser.json());

// ðŸ”¹ Metrics middleware
app.use((req, res, next) => {
  res.on('finish', () => {
    httpRequestCounter.inc({
      method: req.method,
      route: req.route?.path || req.path,
      status: res.statusCode,
    });
  });
  next();
});

// Routes
app.use('/api/auth', authRoutes);
app.use('/api/users', userRoutes);

// ðŸ”¹ Metrics endpoint
app.get('/metrics', async (req, res) => {
  res.set('Content-Type', register.contentType);
  res.end(await register.metrics());
});

// ðŸ”¹ Function to wait until MySQL is ready
const waitForDb = async (retries = 30, delay = 2000) => {
  while (retries > 0) {
    try {
      const [rows] = await db.promise().query(
        "SHOW TABLES LIKE 'users'"
      );

      if (rows.length > 0) {
        console.log('âœ… MySQL users table found.');
        return;
      }

      console.log(
        `â³ Waiting for MySQL (users table)... Retries left: ${retries}`
      );
    } catch (err) {
      console.error(
        `ðŸ”´ MySQL query failed: ${err.message}`
      );
    }

    retries--;
    await new Promise(res => setTimeout(res, delay));
  }

  throw new Error(
    'âŒ MySQL users table not available after multiple retries.'
  );
};

// ðŸ”¹ Function to seed admin user if not exists
const seedAdminUser = async () => {
  const name = process.env.ADMIN_NAME || 'Admin User';
  const email = process.env.ADMIN_EMAIL || 'admin@example.com';
  const password = process.env.ADMIN_PASSWORD || 'admin123';
  const role = process.env.ADMIN_ROLE || 'admin';

  try {
    const [existing] = await db.promise().query(
      'SELECT * FROM users WHERE email = ?',
      [email]
    );

    if (existing.length === 0) {
      const hashed = await bcrypt.hash(password, 10);

      await db.promise().query(
        'INSERT INTO users (name, email, password, role) VALUES (?, ?, ?, ?)',
        [name, email, hashed, role]
      );

      console.log(`âœ… Admin user created â†’ ${email}`);
    } else {
      console.log(`â„¹ï¸ Admin user already exists â†’ ${email}`);
    }
  } catch (err) {
    console.error(
      `âŒ Admin seeding failed: ${err.message}`
    );
  }
};

// ðŸ”¹ Start server
(async () => {
  try {
    await waitForDb();
    await seedAdminUser();

    app.listen(PORT, () => {
      console.log(
        `ðŸš€ Server running on http://0.0.0.0:${PORT}`
      );
    });
  } catch (err) {
    console.error(err.message);
    process.exit(1);
  }
})();
